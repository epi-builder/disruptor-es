pub struct CommandMetadata {
    pub command_id: uuid::Uuid,
    pub correlation_id: uuid::Uuid,
    pub causation_id: Option<uuid::Uuid>,
    pub tenant_id: TenantId,
    pub requested_at: time::OffsetDateTime,

pub struct CommandEnvelope<A: Aggregate> {
    pub command: A::Command,
    pub metadata: CommandMetadata,
    pub idempotency_key: String,
    pub stream_id: StreamId,
    pub partition_key: PartitionKey,
    pub expected_revision: ExpectedRevision,
    pub reply: CommandReply<A::Reply>,
}

impl<A: Aggregate> CommandEnvelope<A> {
    pub fn new(command: A::Command, metadata: CommandMetadata, idempotency_key: impl Into<String>, reply: CommandReply<A::Reply>) -> RuntimeResult<Self>;
}

pub struct CommandGateway<A: Aggregate>;
impl<A: Aggregate> CommandGateway<A> {
    pub fn try_submit(&self, envelope: CommandEnvelope<A>) -> RuntimeResult<()>;

    pub reply: R,
    pub append: es_store_postgres::CommittedAppend,
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP client -> adapter DTOs | Untrusted JSON, tenant IDs, idempotency keys, and metadata enter typed Rust code. |
| Adapter -> CommandGateway | Validated command envelopes cross into bounded runtime ingress. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-01 | Spoofing | `PlaceOrderRequest.tenant_id` and metadata DTOs | mitigate | Construct `TenantId` through `TenantId::new`; never accept tenant from stream ID; include tenant in `CommandMetadata`. |
| T-07-02 | Tampering | idempotency key and command DTOs | mitigate | Reject empty idempotency keys; use domain ID constructors before `CommandEnvelope::new`. |
| T-07-03 | Denial of Service | Tower and `CommandGateway::try_submit` ingress | mitigate | Use nonblocking `try_submit`; map overload to HTTP 429; preserve bounded runtime queue behavior. |
| T-07-04 | Elevation of Privilege | adapter direct state mutation | mitigate | Boundary tests forbid storage/projector/outbox dependencies and hot-state mutation imports. |
</threat_model>
